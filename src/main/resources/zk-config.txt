(4).lcp_framework只存放lcp平台相关的参数，全部加载在JVM中，定时刷新，所以可以放到一个资源占用很小的db中.
create /lcp/dev/mysql/master/framework {"url":"jdbc:mysql://123.57.204.187:3306/lcp_framework?useUnicode=true&characterEncoding=utf-8","userName":"root","userName":"root","password":"111111","driverClassName":"com.mysql.jdbc.Driver"}
